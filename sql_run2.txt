Q: Recommend a highly-rated game (8 or above) that neither Susanne Vejdemo nor Gertrude Young owns, found through a three-hop designer chain: start from a game that Gertrude rates 8 or higher, follow the designer to another game they made, then follow a co-designer on that second game to find the final recommendation. Show the chain: Gertrude's game, the linking designer, the bridge game, the co-designer, and the recommended game.

Generated SQL:
WITH
-- Step 1: Games Gertrude rates >= 8
gertrude_highly_rated AS (
    SELECT pop.game_id, pop.rating AS gertrude_rating
    FROM player_opinions pop
    JOIN players p ON p.id = pop.player_id AND p.label = 'Gertrude Young'
    WHERE pop.rating >= 8
),
-- Step 2: Designers of those games
designers_of_gertrude_games AS (
    SELECT ghr.game_id AS gertrude_game_id, gc.creator_id AS designer_id
    FROM gertrude_highly_rated ghr
    JOIN game_creators gc ON gc.game_id = ghr.game_id
),
-- Step 3: Other games by those designers (bridge games)
bridge_games AS (
    SELECT dogr.gertrude_game_id, dogr.designer_id, gc.game_id AS bridge_game_id
    FROM designers_of_gertrude_games dogr
    JOIN game_creators gc ON gc.creator_id = dogr.designer_id
    WHERE gc.game_id != dogr.gertrude_game_id
),
-- Step 4: Co-designers on bridge games
co_designers AS (
    SELECT bg.gertrude_game_id, bg.designer_id, bg.bridge_game_id, gc.creator_id AS co_designer_id
    FROM bridge_games bg
    JOIN game_creators gc ON gc.game_id = bg.bridge_game_id
    WHERE gc.creator_id != bg.designer_id
),
-- Step 5: Games by co-designers, rated >= 8, not owned by Susanne or Gertrude
recommendations AS (
    SELECT
        cd.gertrude_game_id,
        cd.designer_id,
        cd.bridge_game_id,
        cd.co_designer_id,
        gc.game_id AS rec_game_id
    FROM co_designers cd
    JOIN game_creators gc ON gc.creator_id = cd.co_designer_id
    JOIN games g ON g.id = gc.game_id
    WHERE g.rating >= 8
      AND gc.game_id != cd.bridge_game_id
      AND gc.game_id != cd.gertrude_game_id
      AND NOT EXISTS (
          SELECT 1 FROM player_owns po
          JOIN players p ON p.id = po.player_id AND p.label = 'Susanne Vejdemo'
          WHERE po.game_id = gc.game_id
      )
      AND NOT EXISTS (
          SELECT 1 FROM player_owns po
          JOIN players p ON p.id = po.player_id AND p.label = 'Gertrude Young'
          WHERE po.game_id = gc.game_id
      )
)
SELECT
    g1.name          AS gertrude_game,
    c1.label         AS linking_designer,
    g2.name          AS bridge_game,
    c2.label         AS co_designer,
    g3.name          AS recommended_game,
    g3.rating        AS rec_rating
FROM recommendations r
JOIN games g1 ON g1.id = r.gertrude_game_id
JOIN creators c1 ON c1.id = r.designer_id
JOIN games g2 ON g2.id = r.bridge_game_id
JOIN creators c2 ON c2.id = r.co_designer_id
JOIN games g3 ON g3.id = r.rec_game_id
ORDER BY g3.rating DESC
LIMIT 20;
