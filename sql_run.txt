Q: Recommend a highly-rated game (8 or above) that neither Susanne Vejdemo nor Gertrude Young owns, found through a three-hop designer chain: start from a game that Gertrude rates 8 or higher, follow the designer to another game they made, then follow a co-designer on that second game to find the final recommendation. Show the chain: Gertrude's game, the linking designer, the bridge game, the co-designer, and the recommended game.

Generated SQL:
WITH gertrude_high AS (
    SELECT po.game_id, p.id AS player_id
    FROM player_opinions po
    JOIN players p ON p.id = po.player_id AND p.label = 'Gertrude Young'
    WHERE po.rating >= 8
),
-- Step 1: Get designers of games Gertrude rates 8+
step1 AS (
    SELECT gh.game_id AS gertrude_game_id, gc.creator_id AS linking_designer
    FROM gertrude_high gh
    JOIN game_creators gc ON gc.game_id = gh.game_id
),
-- Step 2: Find other games by that designer (bridge games)
step2 AS (
    SELECT s1.gertrude_game_id, s1.linking_designer, gc.game_id AS bridge_game_id
    FROM step1 s1
    JOIN game_creators gc ON gc.creator_id = s1.linking_designer
    WHERE gc.game_id != s1.gertrude_game_id
),
-- Step 3: Find co-designers on the bridge game (different from linking designer)
step3 AS (
    SELECT s2.gertrude_game_id, s2.linking_designer, s2.bridge_game_id, gc.creator_id AS co_designer
    FROM step2 s2
    JOIN game_creators gc ON gc.game_id = s2.bridge_game_id
    WHERE gc.creator_id != s2.linking_designer
),
-- Step 4: Find other games by the co-designer as final recommendations
step4 AS (
    SELECT s3.gertrude_game_id, s3.linking_designer, s3.bridge_game_id, s3.co_designer, gc.game_id AS rec_game_id
    FROM step3 s3
    JOIN game_creators gc ON gc.creator_id = s3.co_designer
    WHERE gc.game_id != s3.bridge_game_id
      AND gc.game_id != s3.gertrude_game_id
),
-- Filter: recommended game rated 8+, not owned by Susanne or Gertrude
final AS (
    SELECT
        s4.gertrude_game_id,
        s4.linking_designer,
        s4.bridge_game_id,
        s4.co_designer,
        s4.rec_game_id,
        g_rec.rating AS rec_rating
    FROM step4 s4
    JOIN games g_rec ON g_rec.id = s4.rec_game_id
    WHERE g_rec.rating >= 8
      AND NOT EXISTS (
          SELECT 1 FROM player_owns po
          JOIN players p ON p.id = po.player_id AND p.label = 'Susanne Vejdemo'
          WHERE po.game_id = s4.rec_game_id
      )
      AND NOT EXISTS (
          SELECT 1 FROM player_owns po
          JOIN players p ON p.id = po.player_id AND p.label = 'Gertrude Young'
          WHERE po.game_id = s4.rec_game_id
      )
)
SELECT
    g1.name  AS gertrude_game,
    c1.label AS linking_designer,
    g2.name  AS bridge_game,
    c2.label AS co_designer,
    g3.name  AS recommended_game,
    f.rec_rating
FROM final f
JOIN games g1 ON g1.id = f.gertrude_game_id
JOIN creators c1 ON c1.id = f.linking_designer
JOIN games g2 ON g2.id = f.bridge_game_id
JOIN creators c2 ON c2.id = f.co_designer
JOIN games g3 ON g3.id = f.rec_game_id
ORDER BY f.rec_rating DESC
LIMIT 20;
